<script>

const grid = document.getElementById('numberGrid');

const formLink = "https://bit.ly/Sorteo-Nerium";



for (let i = 0; i < 100; i++) {

const numberDiv = document.createElement('div');

numberDiv.classList.add('number');

numberDiv.textContent = i.toString().padStart(2, '0');

numberDiv.setAttribute('aria-label', `Número ${i}`);

numberDiv.addEventListener('click', () => {

const confirmation = confirm(`Has seleccionado el número ${numberDiv.textContent}. ¿Deseas continuar al formulario?`);

if (confirmation) {

numberDiv.classList.add('selected');

window.location.href = formLink;

}

});

grid.appendChild(numberDiv);

}

</script>
